cmake_minimum_required(VERSION 3.15)
project(llama_test)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# 添加 llama.cpp 的路径
set(LLAMA_PATH ${CMAKE_CURRENT_SOURCE_DIR}/llama.cpp)
set(GGML_PATH ${LLAMA_PATH}/ggml)

# 包含头文件
include_directories(${LLAMA_PATH})
include_directories(${LLAMA_PATH}/include)
include_directories(${GGML_PATH}/include)

# 添加 llama.cpp 的源文件
file(GLOB LLAMA_SOURCES 
    ${LLAMA_PATH}/src/*.cpp
    ${GGML_PATH}/src/*.cpp
)

# 创建测试程序
add_executable(test_llama test_llama.cpp ${LLAMA_SOURCES})

# 链接线程库
find_package(Threads REQUIRED)
target_link_libraries(test_llama Threads::Threads)
